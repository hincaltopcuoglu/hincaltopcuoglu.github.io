define shannon entropy

shannon_entropy <- function(x){
    df = count(x)
    df$probs = df$freq / length(df)
    for (i in 1:nrow(df)){
      se<- -sum(df$probs[i] * log(df$probs[i]))
      return(se)
    }
}

play it

shannon_entropy(discrete_normal_arr)
